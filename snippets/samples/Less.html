/* Code for creating CSS hexagons */
@import "constants.less";

@defaultHexColor: @lightBlack;

@defaultHexWidth: 45;
@defaultHexHeight: ceil(@defaultHexWidth / sqrt(3));
@defaultHexSide: @defaultHexHeight - @defaultHexWidth / 13;
@defaultBorderTop: ceil(@defaultHexHeight / 2);
 
.hex {
    .top {
        width: 0;
        border-bottom: @defaultBorderTop solid @defaultHexColor;
        border-left: @defaultHexSide solid transparent;
        border-right: @defaultHexSide solid transparent;
    }
    .middle {
        width: @defaultHexWidth;
        height: @defaultHexHeight;
        background: @defaultHexColor;
    }
    .bottom {
        width: 0;
        border-top: @defaultBorderTop solid @defaultHexColor;
        border-left: @defaultHexSide solid transparent;
        border-right: @defaultHexSide solid transparent;
    }
}
