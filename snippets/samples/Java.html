public Pixel[] grayScale(Pixel[] image) {
  for (Pixel p : image) {
    int pixelAverage = (p.getR() + p.getG() + p.getB()) / 3;
	p.setR(pixelAverage);
	p.setG(pixelAverage);
	p.setB(pixelAverage);
  }
  return image;
}
	
public Pixel[] motionBlur(Pixel[] image, Pixel[] image_copy, int n) {
  for (int i = 0; i < image.length; i++) {
	Pixel p = image[i];
	int r = 0, g = 0, b = 0;
	int end = n;
	int num = n;
	if (end + i % width >= width) {
	  end = width - i % width - 1;
	  num = width - i % width;
	}
	while (--end >= 0) {
	  r += image_copy[i + end].getR();
	  g += image_copy[i + end].getG();
	  b += image_copy[i + end].getB();
	}
	p.setR(r/num);
	p.setG(g/num);
	p.setB(b/num);
  }
  return image;
}
