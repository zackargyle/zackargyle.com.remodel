// Magic
var mousedown = false; 

// Randomly assign a background color to body
$("#easterEgg").mousedown(function() {
  mousedown = true;

  $(document).keypress(function(e) {
    if (mousedown && e.which === ENTER) { 
      var rgb = [];

      // Get three random numbers
      for (var i = 0; i < 3; i++) {
        rgb[i] = Math.floor((Math.random() * 255) + 1);
      }

      $('body').css('background-color','rgb('+rgb[0]+','+rgb[1]+','+rgb[2]+')');
    }
  });
})
.mouseup(function() {
    mousedown = false;
})
.mouseout(function() {
    mousedown = false;
});
